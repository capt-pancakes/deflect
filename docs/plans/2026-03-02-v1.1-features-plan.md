```typescript
export type TutorialAction = 'spawn_signal' | 'spawn_color_signal' | 'complete' | 'wrong_port' | null;

export class TutorialManager {
  phase = 0; // 0=not started, 1=waiting for swipe, 2=signal deflected, 3=color matching, 4=done
  timer = 0;
  ghostSwipeAnim = 0;

  constructor(hasPlayedBefore: boolean) {
    this.reset(hasPlayedBefore);
  }

  reset(hasPlayedBefore: boolean): void {
    if (hasPlayedBefore) {
      this.phase = 4; // Skip tutorial
    } else {
      this.phase = 1;
    }
    this.timer = 0;
    this.ghostSwipeAnim = 0;
  }

  isActive(): boolean {
    return this.phase < 4;
  }

  update(dt: number): void {
    this.timer += dt;
    this.ghostSwipeAnim += dt;
  }

  onSwipe(): void {
    if (this.phase === 1) {
      this.phase = 2;
      this.timer = 0;
    }
  }

  /** Called when the tutorial signal is deflected (bounces off wall). */
  onDeflect(): void {
    if (this.phase === 2) {
      this.phase = 3;
      this.timer = 0;
    }
  }

  /** Phase 2 completion: signal caught or timeout. */
  checkCompletion(signalsRemaining: number): TutorialAction {
    if (this.phase === 2 && (signalsRemaining === 0 || this.timer > 3)) {
      // Transition to phase 3 instead of completing
      this.phase = 3;
      this.timer = 0;
      return 'spawn_color_signal';
    }
    return null;
  }

  /** Phase 3 completion: correct color catch or timeout. */
  checkColorCompletion(correctCatch: boolean): TutorialAction {
    if (this.phase !== 3) return null;
    if (correctCatch) {
      this.phase = 4;
      return 'complete';
    }
    if (this.timer > 8) {
      this.phase = 4;
      return 'complete';
    }
    if (!correctCatch && this.timer > 0) {
      return 'wrong_port';
    }
    return null;
  }
}
```

```typescript
  canvas.width = WIDTH;
```

```typescript
export async function shareScore(text: string, imageCanvas?: HTMLCanvasElement): Promise<boolean> {
  // Try image sharing first
  if (imageCanvas && navigator.share) {
    try {
      const blob = await new Promise<Blob | null>((resolve) =>
        imageCanvas.toBlob(resolve, 'image/png'),
      );
      if (blob) {
        const file = new File([blob], 'deflect-score.png', { type: 'image/png' });
        if (navigator.canShare?.({ files: [file] })) {
          await navigator.share({ text, files: [file] });
          return true;
        }
      }
    } catch {
      // Fall through to text sharing
    }
  }

  // Text sharing fallback
  if (navigator.share) {
    try {
      await navigator.share({ text });
      return true;
    } catch {
      // User cancelled
    }
  }

  try {
    await navigator.clipboard.writeText(text);
    return true;
  } catch {
    return false;
  }
}
```

```typescript
let enabled = true;

export function setHapticsEnabled(on: boolean): void {
  enabled = on;
}

function vibrate(pattern: number | number[]): void {
  if (!enabled) return;
  if (typeof navigator !== 'undefined' && typeof navigator.vibrate === 'function') {
    navigator.vibrate(pattern);
  }
}

export function vibrateOnCatch(): void {
  vibrate(15);
}

export function vibrateOnMiss(): void {
  vibrate([10, 50, 10]);
}

export function vibrateOnMilestone(): void {
```

```typescript
export class PwaPrompt {
  private gameOverCount = 0;
  private hasPlayedDaily = false;
  private dismissed = false;
  private isStandalone = false;
  private deferredPrompt: Event | null = null;

  constructor() {
    // Check if already installed
    try {
      this.isStandalone = window.matchMedia('(display-mode: standalone)').matches;
    } catch {}

    // Check if previously dismissed
    try {
      this.dismissed = localStorage.getItem('deflect_pwa_dismissed') === '1';
    } catch {}

    // Listen for install prompt
    try {
      window.addEventListener('beforeinstallprompt', (e: Event) => {
        e.preventDefault();
        this.deferredPrompt = e;
      });
    } catch {}
  }

  onGameOver(mode: string): void {
    this.gameOverCount++;
    if (mode === 'daily') this.hasPlayedDaily = true;
  }

  shouldShow(): boolean {
    if (this.dismissed || this.isStandalone) return false;
    return this.gameOverCount >= 3 || this.hasPlayedDaily;
  }

  dismiss(): void {
    this.dismissed = true;
    try {
      localStorage.setItem('deflect_pwa_dismissed', '1');
    } catch {}
  }

  async install(): Promise<boolean> {
    if (!this.deferredPrompt) return false;
    try {
      // eslint-disable-next-line @typescript-eslint/no-explicit-any
      const promptEvent = this.deferredPrompt as any;
      promptEvent.prompt();
      const result = await promptEvent.userChoice;
      this.deferredPrompt = null;
      if (result.outcome === 'accepted') {
        this.dismissed = true;
        return true;
      }
    } catch {}
    return false;
  }
}
```

```typescript
import { mutation, query } from './_generated/server';
import { v } from 'convex/values';

export const submitScore = mutation({
  args: { seed: v.number(), score: v.number() },
  handler: async (ctx, { seed, score }) => {
    await ctx.db.insert('dailyScores', { seed, score });
  },
});

export const getStats = query({
  args: { seed: v.number() },
  handler: async (ctx, { seed }) => {
    const scores = await ctx.db
      .query('dailyScores')
      .withIndex('by_seed', (q) => q.eq('seed', seed))
      .collect();
    if (scores.length === 0) return { attempts: 0, percentile: 0 };
    return { attempts: scores.length, percentile: 0 }; // percentile calculated with player score
  },
});

export const getStatsWithPercentile = query({
  args: { seed: v.number(), playerScore: v.number() },
  handler: async (ctx, { seed, playerScore }) => {
    const scores = await ctx.db
      .query('dailyScores')
      .withIndex('by_seed', (q) => q.eq('seed', seed))
      .collect();
    if (scores.length === 0) return { attempts: 0, percentile: 100 };
    const below = scores.filter((s) => s.score < playerScore).length;
    const percentile = Math.round((below / scores.length) * 100);
    return { attempts: scores.length, percentile };
  },
});
```

```typescript
import { ConvexHttpClient } from 'convex/browser';
import { api } from '../convex/_generated/api';

let client: ConvexHttpClient | null = null;

function getClient(): ConvexHttpClient | null {
  if (client) return client;
  const url = import.meta.env.VITE_CONVEX_URL as string | undefined;
  if (!url) return null;
  client = new ConvexHttpClient(url);
  return client;
}

export async function submitDailyScore(seed: number, score: number): Promise<void> {
  const c = getClient();
  if (!c) return;
  try {
    await c.mutation(api.dailyScores.submitScore, { seed, score });
  } catch {
    // Silently fail — game works offline
  }
}

export async function getDailyStats(
  seed: number,
  playerScore: number,
): Promise<{ attempts: number; percentile: number } | null> {
  const c = getClient();
  if (!c) return null;
  try {
    return await c.query(api.dailyScores.getStatsWithPercentile, { seed, playerScore });
  } catch {
    return null;
  }
}
```
